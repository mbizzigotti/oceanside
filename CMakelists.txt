cmake_minimum_required(VERSION 3.0)
project(palomar)

set(CMAKE_C_STANDARD 99)
set(CMAKE_CXX_STANDARD 20)

file(GLOB_RECURSE SOURCE_FILES
	src/*.cpp
	src/3rdparty/cJSON.c
)

add_executable(palomar ${SOURCE_FILES})

# Enable extra warnings
if(MSVC)
	target_compile_options(palomar PRIVATE /W4)
else()
	target_compile_options(palomar PRIVATE -Wall -Wextra -Wpedantic)
endif()

target_link_libraries(palomar PUBLIC
	${CMAKE_SOURCE_DIR}/lib/windows/vulkan.lib
)

if(NOT NOMP)
	if(MSVC)
		target_compile_options(palomar PRIVATE "/MP")
	endif()
endif()
